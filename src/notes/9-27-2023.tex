\section{Variation of Parameters Method}\label{sec:variation-of-parameters-method}
\[ y'' + P(x)y' + Q(x)y = f(x) \] will only work on problems where $P(x)$ and $Q(x)$ are constants.

\subsection{1st Step: General solution of complementary DE}
\[ y = c_{1}y_{1} + c_{2}y_{2} \]
Guess a solution to the non-homogeneous of the form \[ y = u_{1}(x)y_{1}(x) + u_{2}(x)y_{2}(x) \] where $u_{1}$ and $u_{2}$ are functions of $x$.\\\\

This theory produces \[ u_{1}' = \frac{W_{1}}{W} \mbox{ and } u_{2}' = \frac{W_{2}}{W} \] where \[ W = \bigg| \begin{array}{cc}
	y_{1} & y_{2}\\
	y_{1}' & y_{2}'
\end{array} \bigg|,\\
W_{1} = \bigg| \begin{array}{cc}
	0 & y_{2}\\
	f(x) & y_{2}'
\end{array} \bigg|,\\
W_{2} = \bigg| \begin{array}{cc}
	y_{1} & 0\\
	y_{1}' & f(x)
\end{array} \bigg| \]

\example
\[ 4y'' + 36y = \csc(3x) \]
\begin{equation*}
\begin{aligned}
	4y'' + 36y &= \csc(3x)\\
	y'' + 9y &= \frac{\csc(3x)}{4}\\
	m^{2}e^{mx} + 9e^{mx} &= 0\\
	m^{2} + 9 &= 0\\
	m^{2} &= -9\\
	m &= \pm\sqrt{-9}\\
	  &= \pm3i\\
\end{aligned}
\end{equation*}\begin{equation*}
\begin{aligned}
	y_{1} = e^{0x}\cos(3x) \sep y_{2} = e^{0x}\sin(3x)\\
	y_{1} = 1\cos(3x) \sep y_{2} = 1\sin(3x)\\
	y_{1} = \cos(3x) \sep y_{2} = \sin(3x)\\
\end{aligned}
\end{equation*}\begin{equation*}
\begin{aligned}
	y_{c} &= c_{1}\cos(3x) + c_{2}\sin(3x)\\
\end{aligned}
\end{equation*}

Guess \[ y_{p} = u_{1}y_{1} + u_{2}y_{2} \]
\begin{equation*}
\begin{aligned}
	u_{1}' = \frac{W_{1}}{W} \sep u_{2}' = \frac{W_{2}}{W}\\
\end{aligned}
\end{equation*} where \begin{equation*}
\begin{aligned}
	W &= \bigg| \begin{array}{cc}
		\cos(3x) & \sin(3x)\\
		-3\sin(3x) & 3\cos(3x)\\
	\end{array} \bigg|\\
	  &= (3\cos(3x))(\cos(3x)) - (\sin(3x))(-3\sin(3x))\\
	  &= 3\cos^{2}(3x) + 3\sin^{2}(3x)\\
	  &= 3\left( \cos^{2}(3x) + \sin^{2}(3x) \right)\\
	  &= 3\left( 1 \right)\\
	  &= 3\\
\end{aligned}
\end{equation*}
\begin{equation*}
\begin{aligned}
	W_{1} &=\ \bigg| \begin{array}{cc}
		0 & \sin(3x)\\
		\frac{1}{4}\csc(3x) & \cos(3x)\\
	\end{array} \bigg|\\
		  &= 0\cos(3x) - \sin(3x)\left( \frac{\csc(3x)}{4} \right)\\
		  &= -\frac{\sin(3x)\csc(3x)}{4}\\
		  &= -\frac{1}{4}\\
\end{aligned}
\end{equation*}
\begin{equation*}
\begin{aligned}
	W_{2} &=\ \bigg| \begin{array}{cc}
		\cos(3x) & 0 \\
		\sin(3x) & \frac{1}{4}\csc(3x) \\
	\end{array} \bigg|\\
		  &= \frac{1}{4}\csc(3x)\cos(3x) - 0\sin(3x)\\
		  &= \frac{\cos(3x)}{4\sin(3x)}\\
		  &= \frac{1}{4}\cot(3x)\\
\end{aligned}
\end{equation*}\begin{equation*}
\begin{aligned}
	u_{1}' = \frac{W_{1}}{W} \sep u_{2}' = \frac{W_{2}}{W}\\
	u_{1}' = \frac{-\frac{1}{4}}{3} \sep u_{2}' = \frac{\frac{1}{4}\cot(3x)}{3}\\
	u_{1}' = -\frac{1}{12} \sep u_{2} = \frac{1\cot(3x)}{12}\\
	u_{1}' = -\frac{1}{12} \sep u_{2} = \frac{1\cos(3x)}{12\sin(3x)}\\
	u_{1} = \int -\frac{1}{12} dx \sep u_{2} = \int \frac{1\cos(3x)}{12\sin(3x)}dx\\
	u_{1} = -\frac{x}{12} \sep u_{2} = \frac{1}{12}\int \frac{1}{3}\frac{dv}{v}\\
						  \sep u_{2} = \frac{1}{36}\ln|v|\\
						  \sep u_{2} = \frac{1}{36}\ln|\sin(3x)|\\
\end{aligned}
\end{equation*}\begin{equation*}
\begin{aligned}
	y_{p} &= u_{1}y_{1} + u_{2}y_{2}\\
		  &= -\frac{x}{12}\cos(3x) + \frac{1}{36}\ln|\sin(3x)|\sin(3x)\\
		  &= -\frac{x\cos(3x)}{12} + \frac{\sin(3x)}{36}\ln|\sin(3x)|\\
\end{aligned}
\end{equation*}\begin{equation*}
\begin{aligned}
	y &= y_{c} + y_{p}\\
	  &= c_{1}\cos(3x) + c_{2}\sin(3x) - \frac{x\cos(3x)}{12} + \frac{\sin(3x)}{36}\ln|\sin(3x)|\\
\end{aligned}
\end{equation*}

\subsection{3$\times$3 Determinants}
\begin{equation*}
\begin{aligned}
	\bigg| \begin{array}{ccc}
		a & b & c\\
		d & e & f\\
		g & h & i\\
	\end{array} \bigg| = a\bigg| \begin{array}{cc}
		e & f\\
		h & i
	\end{array} \bigg| - b\bigg| \begin{array}{cc}
		d & f\\
		f & i
	\end{array} \bigg| + c\bigg| \begin{array}{cc}
		d & e\\
		g & h
	\end{array} \bigg|
\end{aligned}
\end{equation*}
