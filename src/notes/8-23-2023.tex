\section{Initial Value Problems (IVP)}\label{sec:initial-value-problems}
1st order IVP is a 1st order D.E. together with one extra condition: \[ \frac{dy}{dx}=f(x,y), y(x_{0})=y_{0} \]\\
2nd order IVP \[ y''=f(x,y,y') \]
Initial conditions:
\begin{itemize}
	\item $y(x_{0})=y_{0}$
	\item $y'(x_{0})=y_{1}$
\end{itemize}

\subsection{Example}
\[ y'=y\mbox{ and }y(0)=3\]

\textcolor{orange}{$y=ce^{x}$ is a one-parameter family of solutions \[ \frac{d}{dx}(ce^{x})=ce^{x}=y \]}	% TODO: Change this to a dark green

\begin{equation*}
\begin{aligned}
	ce^{1}&=-2\\
	c&=-\frac{2}{e}\\
	y&=\left( -\frac{2}{e} \right)e^{x}\\
	y&=-2e^{x-1}\\
\end{aligned}
\end{equation*}

\subsection{Example}
\[ \mbox{D.E.: } y'+2xy^{2}=0\mbox{ and } y(0)=1 \]
Given that you have the solution: $y=\frac{1}{x^{2}+C}$, Solve:
\begin{equation*}
\begin{aligned}
	-1&=\frac{1}{(0)^{2}+c}\\
	-1&=\frac{1}{c}\\
	-1\times c&=1\\
	c&=-1\\
	y&=\frac{1}{x^{2}-1}, I=(-1, 1)
\end{aligned}
\end{equation*}

\subsubsection{Example}
\[ x''+16x=0\mbox{ and } x(\frac{\pi}{2})=5\mbox{ and } x'(\frac{\pi}{2})=-4 \]
\begin{equation*}
\begin{aligned}
	x&=c_{1}\cos(4t)+c_{2}\sin(4t)\\
	5&=c_{1}\cos(4t)+c_{2}\sin(4t)\\
	 &=c_{1}\cos(2\pi)+c_{2}\sin(2\pi)\\
	 &=c_{1}(1)+c_{2}(0)\\
	 &=c_{1}\\
	x'&=-4c_{1}\sin(4t)+4c_{2}\cos(4t)\\
	-4&=-4c_{1}\sin\left(4\left( \frac{\pi}{2} \right)\right)+4c_{2}\cos\left(4\left( \frac{\pi}{2} \right)\right)\\
	  &=-4c_{1}\sin(2\pi)+4c_{2}\cos(2\pi)\\
	  &=-4c_{1}(0)+4c_{2}(1)\\
	  &=4c_{2}\\
	-1&=c_{2}
\end{aligned}
\end{equation*}

\noindent Reasonable Question: Given a 1st order IVP, can we say whether a solution \emph{exists} or not and, if a solution exists, is it \emph{unique}.\\

\noindent \theorem{Given $y'=f(x,y)$ and $y(x_{0})=y_{0}$, if $f(x,y)$ and $\frac{\partial f}{\partial y}$ are both continuous on a rectangle $R$ containing $(x_{0},y_{0})$ in its interior, then there exists an interval $I=(x_{0}-h,x_{0}+h)$ where $h>0$ such that there exists a unique solution to IVP on $I$.}

\subsection{Example}
\[ y'=xy^{\frac{1}{2}}\mbox{ and } y(1)=2\]
\begin{itemize}
	\item $f(x,y)=xy^{\frac{1}{2}}$ is continuous everywhere its defined $y\geq 0$
	\item $\frac{\partial f}{\partial y}=x\frac{1}{2}y^{-\frac{1}{2}}=\frac{x}{2\sqrt{y}}$ is continuous everywhere its defined $y>0$
\end{itemize}

\subsection{Example}
\[ y'=xy^{\frac{1}{2}}\mbox{ and } y(0)=0\]
\begin{itemize}
	\item $f(x,y)=xy^{\frac{1}{2}}$ is continuous for all $x$ and $y\geq 0$
	\item $\frac{\partial f}{\partial}=\frac{x}{2y}$ is continuous for all $x$ and $y>0$.
	\item \emph{\textbf{Theorem does not give any conclusion.}}
\end{itemize}
