Shifting Theorem Shifting on $t$-axis

\pdfbookmark[4]
{Operational Rule \#2 Equation}
{subsec:operational-rule-2}
\[ \laplace \left\{ f(t-a)\unitstep(t-a) \right\} = e^{-as}\laplace\left\{ f(t) \right\} \]


Proof: \begin{equation*}
\begin{aligned}
	\laplace \left\{ f(t-a)\unitstep(t-a) \right\} &= \int_{0}^{\infty} e^{-st}f(t-a)\unitstep(t-a)\\
												   &= \int_{0}^{a}0dt + \int_{a}^{\infty} e^{-st}f(t-a)dt\\
	\mbox{Let } \tau = t-a,\ d\tau = dt	\ \ \ \    &= 0 + \int_{a}^{\infty} e^{-s(\tau+a)}f(\tau)d\tau\\
												   &= e^{-sa}\int_{a}^{\infty} e^{-s\tau}f(\tau)d\tau\\
												   &= e^{-sa}\laplace\left\{ f(\tau) \right\}\\
\end{aligned}
\end{equation*}\\

\[ \laplace\left\{ f(\tau)\unitstep(\tau) \right\} = e^{-as}\laplace\left\{ f(\tau) \right\} \]
\begin{equation} \laplace\left\{ f(t-a)\unitstep(t-a) \right\} = e^{-as}\laplace\left\{ f(t-a) \right\}\label{eq:laplace-t-shift}
\end{equation}\\

\[ f(t-a)\unitstep(t-a) = \laplace^{-1}\left\{ e^{-as}F(s) \right\} \]
\begin{equation} f(t-a)\unitstep(t-a) = \laplace^{-1}\left\{ e^{-as}\laplace\left\{ f(t) \right\} \right\} \label{eq:laplace-inverse-t-shift}\end{equation}
