\section{Systems of Higher Order Linear DEs}\label{sec:systems-of-higher-order-linear-des}

\[ \mbox{Solve} \left\{ \begin{array}{lllr}
	x' &+ y' &+ 2y &= 0\\
	x' &- 3x &- 2y &= 0\\
\end{array} \right. \]

Need to find $x(t)$ and $y(t)$ that simultaneously solve these equations.

\subsection{Method: Systematic Elimination}\label{subsec:method:-systematic-elimination}
Change from using prime notation to indicate derivatives to the $D$ operator notation. \[ x'\Rightarrow Dx\ \ ,\ \  y' + 2y\Rightarrow (D+2)y \]

\begin{equation*}
\begin{aligned}
	\left\{ \begin{array}{lllr}
		Dx + (D+2)y &= 0\\
		(D-3)x - 2y &= 0\\
	\end{array} \right\} &= \left\{ \begin{array}{lllr}
		(D-3)\left[ Dx + (D+2)y \right] &= 0\\
		D\left[ (D-3)x - 2y \right] &= 0\\
	\end{array} \right\}\\ &= \left\{ \begin{array}{lllr}
		(D^{2} - 3D)x + (D^{2} - D - 6)y &= 0\\
		(D^{2} - 3D)x - 2Dy &= 0\\
	\end{array} \right\}\\ &= \left\{ \begin{array}{lllr}
		x'' - 3x' + y'' - y' - 6y &= 0\\
		x'' - 3x' - 2y' &= 0\\
	\end{array} \right\}\\
	x'' - 3x' + y'' - y' - 6y &= x'' - 3x' - 2y'\\
	y'' - y' - 6y &= -2y'\\
	y'' + y' - 6y &= 0\\
	m^{2}e^{mx} + me^{mx} - 6e^{mx} &= 0\\
	m^{2} + m - 6 &= 0\\
	(m+3)(m-2) &= 0\\
\end{aligned}
\end{equation*}\begin{equation*}
\begin{aligned}
	m_{1} + 3 = 0 \sep m_{2} - 2 = 0\\
	m_{1} = -3 \sep m_{2} = 2\\
	y_{1} = e^{-3t} \sep y_{2} = e^{2t}\\
\end{aligned}
\end{equation*}\begin{equation*}
\begin{aligned}
	y(t) &= c_{1}e^{-3t} + c_{2}e^{2t}
\end{aligned}
\end{equation*}

Using the same ideas, we can find $x(t)$, namely: \begin{equation*}
\begin{aligned}
	\left\{ \begin{array}{lllr}
		Dx + (D+2)y &= 0\\
		(D-3)x - 2y &= 0\\
	\end{array} \right\} &= \left\{ \begin{array}{lllr}
		(-2)\left[ Dx + (D+2)y \right] &= 0\\
		(D+2)\left[ (D-3)x - 2y \right] &= 0\\
	\end{array} \right\}\\ &= \left\{ \begin{array}{lllr}
		-2Dx - (2D+4)y &= 0\\
		(D^{2} - D - 6)x - (2D + 4)y &= 0\\
	\end{array} \right\}\\ &= \left\{ \begin{array}{lllr}
		-2x' &- 2y' - 4y &= 0\\
		x'' - x' - 6x &- 2y' - 4y &= 0\\
	\end{array} \right\}\\
\end{aligned}
\end{equation*}\begin{equation*}
\begin{aligned}
	x'' - x' - 6x - 2y' - 4y &= -2x' - 2y' - 4y\\
	x'' - x' - 6x &= -2x'\\
	x'' + x' - 6x &= 0\\
	n^{2}e^{nt} + ne^{nt} - 6e^{nt} &= 0\\
	n^{2} + n - 6 &= 0\\
	(n+3)(n-2) &= 0\\
\end{aligned}
\end{equation*}\begin{equation*}
\begin{aligned}
	n_{1} + 3 = 0 \sep n_{2} - 2 = 0\\
	n_{1} = -3 \sep n_{2} = 2\\
	x_{1} = e^{-3t} \sep x_{2} = e^{2t}\\
\end{aligned}
\end{equation*}\begin{equation*}
\begin{aligned}
	x(t) &= c_{3}e^{-3t} + c_{4}e^{2t}
\end{aligned}
\end{equation*}

However, the 4 constants $c_{1}$, $c_{2}$, $c_{3}$, $c_{4}$ are not completely independent of each other.
To see their dependency, plug $x(t)$ \& $y(t)$ into each of the original equations.
\begin{equation*}
\begin{aligned}
	c_{3}e^{-3t} - 3c_{1}e^{-3t} + c_{1}e^{-3t} &= 0\\
	c_{4}e^{2t} - 3c_{1}e^{-3t} + c_{1}e^{-3t} &= 0\\
	c_{3}e^{-3t} + 2c_{1}e^{2t} + c_{1}e^{2t} &= 0\\
	c_{4}e^{2t} + 2c_{1}e^{2t} + c_{1}e^{2t} &= 0\\
	\dots\\
	-(3c_{3} + c_{1})e^{-3t} + (2c_{4}+4c_{2})e^{2t} = 0\\
\end{aligned}
\end{equation*}
for this to be true for all $t$, you need \[ \left\{ \begin{array}{ll}
	3c_{3} + c_{1} = 0 &\Rightarrow c_{3} = -\frac{1}{3}c_{1}\\
	2c_{4} + 4c_{2} = 0 &\Rightarrow c_{4} = -2c_{2}\\
\end{array} \right. \]

\example
\[ \left\{ \begin{array}{ll}
	x' - 4x + y'' &= t^{2}\\
	x' +  x + y'  &= 0\\
\end{array} \right. \]
\begin{equation*}
\begin{aligned}
	\left\{ \begin{array}{ll}
		x' - 4x + y'' &= t^{2}\\
		x' +  x + y'  &= 0\\
	\end{array} \right\} &= \left\{ \begin{array}{ll}
	(D-4)x + D^{2}y &= t^{2}\\
	(D+1)x + Dy  &= 0\\
\end{array} \right\}\\
	&= \left\{ \begin{array}{ll}
		(D-4)x + D^{2}y &= t^{2}\\
		D(D+1)x + D^{2}y  &= 0\\
	\end{array} \right\}\\
	&= \left\{ \begin{array}{ll}
		(D-4)x + D^{2}y &= t^{2}\\
		(D^{2}+D)x + D^{2}y  &= 0\\
	\end{array} \right\}\\
\end{aligned}
\end{equation*}
\begin{equation*}
\begin{aligned}
	(D-4)x + D^{2}y - (D^{2}+D)x - D^{2}y &= t^{2} - 0\\
	x' - 4x - x'' - x' &= t^{2} - 0\\
	- 4x - x'' &= t^{2}\\
	x'' + 4x &= -t^{2}\\
\end{aligned}
\end{equation*}

\subsubsection{First Solve $x'' + 4x = 0$}
Guess $x = e^{mt}$
\begin{equation*}
\begin{aligned}
	m^{2} + 4 &= 0\\
	m^{2} &= -4\\
	m &= \pm\sqrt{-4}\\
	m &= \pm2i\\
	x(t) &= c_{1}\cos(2t) + c_{2}\sin(2t)
\end{aligned}
\end{equation*}
